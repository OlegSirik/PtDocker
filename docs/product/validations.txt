 rule_service
    service based on CEL common expression language
    load all rules and compile them in cache

    create all CRUD methods for pt_rules

    method process_validation(ruleType, scopeCode - product_code, ctx context) return [String]
        ruleType must be in PRE_QUOTE_VALIDATION, POST_QUOTE_VALIDATION, PRE_SAVE_VALIDATION, POST_SAVE_VALIDATION
        
        select rules from pt_product_versions where ruleType = ruleType order by priority
        execute expression
        if result != FALSE then add message to [String]

        select rules from pt_rules where ruleType = ruleType and scopeType = LOB and scopeCode = lob_code order by priority
        execute expression
        if result != FALSE then add message to [String]

        select rules from pt_rules where ruleType = ruleType and scopeType = TENANT order by priority
        execute expression
        if result != FALSE then add message to [String]

        select rules from pt_rules where ruleType = ruleType and scopeType = CLIENT order by priority
        execute expression
        if result != FALSE then add message to [String]

        return [String]

    other validation will be added in next release
